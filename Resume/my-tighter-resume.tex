\resheading{Significant Projects}
\begin{itemize}

	\begin{project}{Department Website}{Calvin College}{Sole Programmer}{January 2007--present}
	  \vspace{-1mm}
	  \begin{descriptio}
	  	The Computer Science department's \href{http://cs.calvin.edu/}{website} at Calvin College is a Ruby-on-Rails webapp (source on \href{http://www.github.com/jdfrens/calvincs}{GitHub}).  The app provides a modest CMS with news and events as well as specialized features for an academic institution (e.g., faculty details, course information).
	  \end{descriptio}
	  \vspace{-1mm}
	  \begin{itemize}
	    \resitem{Used as a sandbox for developing course material.}
	    \resitem{Using behavior-driven development with RSpec and Cucumber.}
	  	\resitem{Administering production server.}
	  \end{itemize}
	\end{project}

	\begin{project}{ANTLR Testing, CIAT, CITkit}{SourceForge, GitHub}{Sole Programmer}{Summer 2003--present}
	  \vspace{-1mm}
		\begin{descriptio}
			Three related projects for creating interpreters and compilers.  \href{http://antlr-testing.sourceforge.net/}{ANTLR Testing} is a JUnit extension for testing ANTLR grammars.  \href{http://jdfrens.github.com/ciat/}{CIAT} is a framework for writing acceptance tests for interpreters and compilers, invoked as a rake task.  \href{http://citkit.sourceforge.net/}{CITkit} is a Java library to support the building of interpreters and compilers.
		\end{descriptio}
	\end{project}

	\begin{project}{YAGS}{Calvin College}{Project Lead and Programmer}{Summer 2007}
	  \vspace{-1mm}
		\begin{descriptio}
		  \href{htttp://www.github.com/jdfrens/yags/}{YAGS (Yet Another Genetics Simulator)} is a Ruby-on-Rails webapp that simulates Mendelian genetics (including linked genes and chromosomal crossover) in fruit flies for biology students.
		\end{descriptio}
	  \vspace{-2mm}
		\begin{itemize}
		  \resitem{Used Extreme Programming.}
			\resitem{Managed two student programmers.}
			\resitem{Developed and reviewed code.}
		\end{itemize}
	\end{project}

	\begin{project}{No Latte}{SourceForge}{Sole Programmer}{Summer 2003--present}
	  \vspace{-1mm}
	  \begin{descriptio}
	  	\href{http://nolatte.sourceforge.net/}{No Latte} is an interpreter for a language for writing XHTML documents in a functional-programming style---\LaTeX\ sensibilities with LISP semantics.
	  \end{descriptio}
	  \vspace{-2mm}
		\begin{itemize}
		  \resitem{Testing with mock objects and user-level acceptance tests.}
			\resitem{Implemented in Java; uses ANTLR for the front-end.}
		\end{itemize}
	\end{project}
	
\end{itemize}
